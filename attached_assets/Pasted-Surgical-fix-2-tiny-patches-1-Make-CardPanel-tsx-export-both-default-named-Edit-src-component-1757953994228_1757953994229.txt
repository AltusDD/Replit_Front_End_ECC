Surgical fix (2 tiny patches)
1) Make CardPanel.tsx export both default & named

Edit: src/components/cardkit/CardPanel.tsx

Replace the file with this:

import React from "react";

type CardPanelProps = {
  title?: string;
  children: React.ReactNode;
  testid?: string;
  className?: string;
};

function CardPanel({ title, children, testid, className = "" }: CardPanelProps) {
  return (
    <section
      data-testid={testid}
      className={`rounded-2xl border border-neutral-800 p-4 bg-neutral-950 ${className}`}
    >
      {title && <h3 className="text-sm font-semibold text-neutral-300 mb-3">{title}</h3>}
      <div className="space-y-2">{children}</div>
    </section>
  );
}

export default CardPanel;
export { CardPanel }; // <-- provide named export too


This keeps existing default imports working and unbreaks any import { CardPanel } calls.

2) Do the same for FieldGroup to prevent the next crash

import React from "react";

type FieldGroupProps = {
  label: string;
  value?: React.ReactNode;
  testid?: string;
  className?: string;
};

function FieldGroup({ label, value, testid, className = "" }: FieldGroupProps) {
  return (
    <div data-testid={testid} className={`rounded-xl border border-neutral-800 p-3 ${className}`}>
      <div className="text-xs uppercase tracking-wide text-neutral-500">{label}</div>
      <div className="text-sm text-neutral-200 mt-1">{value ?? "—"}</div>
    </div>
  );
}

export default FieldGroup;
export { FieldGroup }; // <-- provide named export too
